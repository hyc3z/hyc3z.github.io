Script for monitoring learning progress‚Äã

A takeaway python script to encourage yourself :) around 50 lines.

import re
import os
from datetime import datetime
import pandas as pd
import time

def format_diff(diff):
    if diff > 1:
        return "{}s".format(round(diff,2))
    elif diff > 0.001:
        return "{}ms".format(round(diff*1000,2))
    elif diff > 0.000001:
        return "{}us".format(round(diff*1000000, 2))

def summary(books_unread, books_finished):
    total_books_unread = len(books_unread)
    total_books_finished = len(books_finished)
    total_books = total_books_unread + total_books_finished
    percentage_finished = round(total_books_finished / total_books * 100, 2)
    current_year = datetime.now().year
    current_month = datetime.now().month
    df = pd.DataFrame(books_finished, columns = ['book_name', 'finish_date'])
    df["finish_date"] = pd.to_datetime(df["finish_date"], format=r'%b %d %Y')
    df.sort_values(by='finish_date', inplace=True)
    print(df)
    last_book = df.tail(1)
    this_month_finished = len(df[(df["finish_date"] >= '{}-{}-01'.format(current_year, current_month)) & (df["finish_date"] < datetime.now())])
    if total_books > 0:
        print("You've kept {} books in total, in which you finished {}, which is {}%. {}".format(total_books, total_books_finished, percentage_finished, "Great Job!" if percentage_finished > 0 else "Try to keep up!"))
        print("You've finished {} books this month, great progress!".format(this_month_finished) if this_month_finished > 0 else "You've not finished any books this month, keep up!")
        print("Your last finished book is \"{}\" at {}".format(last_book.iloc[0]['book_name'], last_book.iloc[0]['finish_date'].strftime('%Y-%m-%d')))
        if total_books_finished > 0:
            print("There are {} books ahead, have fun reading!".format(total_books_unread))
        else:
            print("There is always something you can learn. Try to hunt for some cool books :)")
    else:
        print("You haven't made your learning plan :)")

def main():
    t0 = time.time()
    path = os.path.join(os.getcwd(), "./misc/masterplan/learning-plan.md")
    with open(path) as f:
        content = f.read()
        pattern_unread_books = "^\s*?- \[ \] (.*?)$"
        pattern_finished_books = "^\s*?- \[x\] (.*?)$\s*^\s*- First pass:\s*(.*?)$"
        total_books_unread = re.findall(pattern_unread_books, content, re.MULTILINE)
        total_books_finished = re.findall(pattern_finished_books, content, re.MULTILINE)        
        t1 = time.time()
        summary(total_books_unread, total_books_finished)
        t2 = time.time()
        print("Read time: {} Summary Time: {}".format(format_diff(t1-t0), format_diff(t2-t1)))

if __name__ == '__main__':
    main()
